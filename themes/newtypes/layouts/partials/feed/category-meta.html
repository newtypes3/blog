{{- /* category-meta.html — context = page Renders timestamp + category label inline (used by both short post and long card). Category color comes from data/categories.toml (user override possible). Site params control timestamp format. */ -}} {{- $p := . -}} {{- $tsFormat := site.Params.timestampFormat | default "2006.01.02 · 15:04" -}} {{- $ts := $p.Date.Format $tsFormat -}} {{- $cats := $p.Params.categories -}} {{- $cat := "" -}} {{- if $cats -}}{{ $cat = index $cats 0 }}{{- end -}} {{ $ts }} {{- if $cat -}} {{- $catUpper := upper $cat -}} {{- $catMap := hugo.Data.categories -}} {{- $catEntry := index $catMap (lower $cat) -}} {{- $color := "var(--fg-2)" -}} {{- if $catEntry -}} {{- with $catEntry.color -}}{{ $color = . }}{{- end -}} {{- end -}} {{ $catUpper }} {{- end -}}
